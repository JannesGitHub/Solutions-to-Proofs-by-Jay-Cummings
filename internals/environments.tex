
% Conjecture environment
\definecolor{conjecture_color_background}{HTML}{E1F7D5} % Heller Hintergrund
\definecolor{conjecture_color_frame}{HTML}{5DBE38} % Dunklerer Rahmen
\definecolor{conjecture_color_title}{HTML}{FFFFFF} % Wei√üer Titel
\newtcolorbox[auto outer sizes, number within=section]{conjecture}[2][]{%
  enhanced,
  sharp corners,
  colback=conjecture_color_background,
  colframe=conjecture_color_frame,
  coltitle=conjecture_color_title,
  boxrule=0.75mm,
  attach boxed title to top left={yshift*=-\tcboxedtitleheight/2,xshift=10mm},
  boxed title style={colback=conjecture_color_frame, sharp corners},
  title=Conjecture #2,
  #1
}
